from Ultilities import xy0
xy0()
def start():
	## Start planting pumpkins
	
	for i in range(6):
		for j in range(5):
		
			if get_entity_type()==None or get_entity_type()==Entities.Dead_Pumpkin:
				plant(Entities.Pumpkin)
					
			else:			# Otherwise, harvest the current crop and plant a pumpkin
				harvest()
				plant(Entities.Pumpkin)
					
			if i%2!=0:
				move(South)
				
			else:	
				move(North)
		plant(Entities.Pumpkin)
		move(East)
			

def noDeadP():
	# Search for dead pumpkins and store their X and Y coordinates 
	XdeadP=[]
	YdeadP=[]
		
	for x in range(get_world_size()):
		for y in range(get_world_size()):
			if get_entity_type()==Entities.Dead_Pumpkin:
					XdeadP.append(get_pos_x())
					YdeadP.append(get_pos_y())
					use_item(Items.Water)
					use_item(Items.Water)
					plant(Entities.Pumpkin) # Replant to try growing a healthy pumpkin
				
				
			if x % 2 == 0:
				move(North)
			else:
				move(South)
		move(East)
		
	#  Visit every recorded dead pumpkin and keep replanting until it becomes a healthy pumpkin
	while 0!= len(XdeadP):
			
			# Repeat for every stored dead pumpkin
			for j in range(len(XdeadP)):
				
				if get_pos_x()>=XdeadP[j]:
					for i in range(get_pos_x()-XdeadP[j]): # Move according to the distance between the drone and the dead pumpkin
						move(West)
				else:	
					for i in range(XdeadP[j]-get_pos_x()):
						move(East)				
				if get_pos_y()<=YdeadP[j]: 
					for i in range(YdeadP[j]-get_pos_y()):
						move(North)
				else:
					for i in range(get_pos_y()-YdeadP[j]):
						move(South)
					
				# If there is no more Dead_Pumpkin in the position, remove his cordinates from the lists 
				if get_entity_type()==Entities.Pumpkin:
					
					XdeadP.pop(j)
					YdeadP.pop(j)
					break
				else:
					use_item(Items.Water)
					use_item(Items.Water)
					plant(Entities.Pumpkin)
					plant(Entities.Pumpkin)
				
 
			
start()			
noDeadP()	
				
harvest()
